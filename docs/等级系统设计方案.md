async function addExperience(qcio_id, source, baseAmount) {
  const $ = db.command.aggregate;

  // è·å–ç”¨æˆ·ä¿¡æ¯
  const userRes = await db.collection('qcio_users').doc(qcio_id).get();
  const user = userRes.data;

  // è®¡ç®—VIPåŠ æˆ
  const vipBonus = user.vip_exp_bonus || 1.0;
  const totalAmount = Math.floor(baseAmount * vipBonus);

  // æ£€æŸ¥æ¯æ—¥é™åˆ¶
  const today = new Date().toDateString();
  const dailyKey = `daily_exp_${source}_${today}`;

  // æ£€æŸ¥æ˜¯å¦è¶…é™

  if (dailyLimit < Infinity) {
    const todayLogs = await db.collection('qcio_experience_logs')
      .where({
        qcio_id,
        action: source,

      .count();

    if (todayLogs.total >= dailyLimit) {
      return { success: false, message: 'ä»Šæ—¥ç»éªŒå·²è¾¾ä¸Šé™' };
    }
  }

  // æ›´æ–°ç”¨æˆ·ç»éªŒ
  const newExp = user.experience + totalAmount;
  const newTotalExp = user.total_experience + totalAmount;

  // æ£€æŸ¥æ˜¯å¦å‡çº§
  let newLevel = user.level;
  let levelUpRewards = [];
  let currentNeedExp = getExpForLevel(newLevel + 1);

  while (newExp >= currentNeedExp) {
    newLevel++;
    currentNeedExp = getExpForLevel(newLevel + 1);

    // æ£€æŸ¥æ˜¯å¦æœ‰ç­‰çº§å¥–åŠ±
    const reward = await db.collection('qcio_level_rewards')
      .where({ level: newLevel })
      .get();

    if (reward.data.length > 0) {
      levelUpRewards.push(reward.data[0]);
    }
  }

  // æ›´æ–°ç”¨æˆ·æ•°æ®
  await db.collection('qcio_users').doc(qcio_id).update({
    experience: newExp,
    total_experience: newTotalExp,
    level: newLevel,

    level_title: getLevelTitle(newLevel),
  });

  // è®°å½•ç»éªŒæ—¥å¿—
  await db.collection('qcio_experience_logs').add({
    data: {

      base_exp: baseAmount,

      total_exp: totalAmount,
      before_level: user.level,
      after_level: newLevel,
      timestamp: new Date(),
    }

    success: true,
    new_level: newLevel,
    level_up: newLevel > user.level,
    level_up_rewards: levelUpRewards,
    experience: totalAmount,
  };
}
